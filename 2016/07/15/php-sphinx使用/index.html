Centos7 Sphinx install

  • 安装依赖库

    yum install make gcc gcc++ gcc-c++ libtool autoconf automake imake libxml2-devel expat-devel
cd /usr/local/src

// https://share.weiyun.com/2da811898545e0043a00c4129fea95c0
// download coreseek-3.2.14
tar xvzf coreseek-3.2.14.tar.gz
cd coreseek-3.2.14
cd mmseg-3.2.14
./bootstrap
./configure --prefix=/usr/local/mmseg3
make && make install
cd /usr/local/src/coreseek-3.2.14/csft-3.2.14

#这里我们要修改 src/sphinxexpr.cpp 文件，将该文件中的 1013、1047、1080 行的 ExprEval 改为 this->ExprEval
sh buildconf.sh

#修改configure文件,找到“#define USE_LIBICONV 1”，将注释去掉，并将1改成0。
make && make install

cd ..
cd testpack
cat var/test/test.xml   #此时应该正确显示中文
/usr/local/mmseg3/bin/mmseg -d /usr/local/mmseg3/etc var/test/test.xml  #整篇文章进行分词
/usr/local/coreseek/bin/indexer -c etc/csft.conf --all  #创建索引
/usr/local/coreseek/bin/search -c etc/csft.conf 网络搜索    #搜索关键字 网络搜索
/usr/local/coreseek/bin/searchd -c etc/csft.conf    #正常开启搜索服务
/usr/local/coreseek/bin/searchd -c etc/csft.conf --stop #如要停止搜索服务
/usr/local/coreseek/bin/indexer -c etc/csft.conf --all --rotate #如要已启动服务，要更新索引

  • 安装PHP扩展

    cd /usr/local/src/coreseek-3.2.14/csft-3.2.14/api/libsphinxclient
./configure --prefix=/usr/local/sphinxclient
make && make install

cd /usr/local/src
wget http://pecl.php.net/get/sphinx-1.1.0.tgz
tar -xzvf sphinx-1.1.0.tgz
cd sphinx-1.1.0
/usr/bin/phpize
./configure --with-php-config=/usr/local/php/bin/php-config --with-sphinx=/usr/local/sphinxclient

make && make install

#注意，如果你的php版本是5.4，那么在这一步中会出现错误，提示在 sphinx.c:105:2，可以按下面方式修改：

#vim sphinx.c，找到105行

#把 retval = std_hnd->read_property(object, member, type TSRMLS_CC); 修改成 retval = std_hnd->read_property(object, member, type TSRMLS_CC, NULL);

# 重新编译
./configure --with-php-config=/usr/local/php/bin/php-config --with-sphinx=/usr/local/sphinxclient
make && make install

# 修改php.ini文件
vim /usr/local/php/etc/php.ini
# 在最后一行加上
extension="sphinx.so"
# 重启
/etc/init.d/php-fpm restart
/etc/init.d/nginx restart

      cd /usr/local/coreseek/etc/

  • 创建数据库，导入测试数据

    在主数据源设置中，sql_query和sql_query_info字段都是From document，这是安装coreseek时提供的测试表，位于/usr/local/coreseek/etc/example.sql

    mysql -uroot -p123456

use test;
source /usr/local/coreseek/etc/example.sql

  • 搜索测试

    • 创建索引，每次新增数据后，都要重新索引一次

    • 创建索引命令indexer

      • -c 指定配置文件
      • –all 对所有索引重新编制索引
      • –rotate 用于轮换索引，主要是在不停止服务的时候增加索引
      • –merge 合并索引

      • 针对test:documents创建索引

        cd /usr/local/coreseek/bin
./indexer -c /usr/local/coreseek/etc/sphinx.conf --all

    • 查询关键字

      • 查询命令search
        • -c 指定配置文件

      • 查关键字test

        cd /usr/local/coreseek/bin
./search -c /usr/local/coreseek/etc/sphinx.conf test

      • 查询关键字group

        ./search -c /usr/local/coreseek/etc/sphinx.conf group

      • 插入一条数据并检索

        mysql -uroot -p123456 test
insert into documents(group_id, group_id2, date_added, title, content) values(3,9,now(),'zhangsan', 'zhangsan is a student');
      • 查询关键字zhangsan

        检索结果为0，因为每次新增数据都需要重新添加索引

      • 插入一条中文书籍并检索

        mysql -uroot -p123456 test
insert into documents(group_id, group_id2, date_added, title, content) values(3,9,now(),'zhangsan张三', '张三是一名学生');

# 重新建立索引
./indexer -c /usr/local/coreseek/etc/sphinx.conf --all

在PHP中使用Sphinx

  • 在php中使用sphinx步骤
    • 建立sphinx配置文件
    • 生成索引
    • 启动sphinx服务进程(searchd)
    • 安装PHP的sphinx扩展
    • 使用php客户端连接sphinx服务器
  • 启动sphinx服务命令searchd
    • -c 指定配置文件
    • –stop 停止服务
    • –pidfile 显式指定一个PID文件
    • -p 指定端口

  • 启动sphinx服务进程(searchd)，并开启端口9312

    cd /usr/local/coreseek/bin
./searchd(或者 ./searchd -c /usr/local/coreseek/etc/csft.conf) # 这里默认会加载csft.conf文件，如果配置文件不叫csft.conf可以使用-c命令指定配置文件名称

  • 查看searchd状态

  • 简单的PHP检索示例

    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Document</title>
</head>
<body>
<?php

    if (empty($_GET['key']) && !isset($_GET['key'])) {
            echo '请输入搜索内容';
            exit();
    }

    define('INDEX', 'test1');

    $sphinx = new SphinxClient();

    // 设置主机名和端口号(默认9312)
    $sphinx->SetServer('localhost', 9312);

    /**
     * 设置匹配方式
     * 1. SPH_MATCH_ANY表示所有结果
     * 2. SPH_MATCH_ALL表示包涵关键词的结果
     */
    $sphinx->SetMatchMode(SPH_MATCH_ANY);

    // query('a', 'b')在索引中搜索关键字a, query('a', '*')在所有索引中搜索a
    $result = $sphinx->query($_GET['key'], INDEX);

    echo '<pre>';
    print_r($result);
    echo '</pre>';

    $ids = join(',', array_keys($result['matches']));
    $mysql = new mysqli('localhost', 'root', '123456', 'test');
    $mysql->query('SET NAMES utf8');
    $sql = "select * from documents where id in({$ids})";

    $opts = array(
        'before_match' => "<font style='font-weight: bold; color: #f00;'>",
        'after_match'  => '</font>'
    );

    $res = $mysql->query($sql);

    $count = $res->num_rows;

    if ($count > 0) {
        while ($list = $res->fetch_assoc()) {
            $row = $sphinx->buildExcerpts($list, INDEX, $_GET['key'], $opts);

            echo "标题：{$row[4]}<br/>";
            echo "内容：{$row[5]}";

            echo '<hr>';
        }

        echo "总共搜索到${count}条";
    } else {
        echo '没有搜索到';
    }
?>
</body>
</html>
