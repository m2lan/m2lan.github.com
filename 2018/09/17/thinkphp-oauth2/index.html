Authorized认证方式

http://swoole-chat.com/index/index/testAuthorize?response_type=code&client_id=phpweb&state=xyz

http://swoole-chat.com/index/index/testToken form-data: [grant_type: authorization_code, code: authorizedCode]

use utils\Crypt;
use OAuth2\Autoloader;
use OAuth2\Server;
use OAuth2\GrantType\AuthorizationCode;
use OAuth2\GrantType\ClientCredentials;
use OAuth2\GrantType\UserCredentials;
use OAuth2\Request;
use OAuth2\Response;
use OAuth2\Storage\Redis as StorageRedis;
use cache\Redis;

header('Access-Control-Allow-Origin:*');

class Index {

    protected $server;

    public function __construct() {

        // 使用PDO方式存储
        // $storage = new Pdo(['dsn' => 'mysql:dbname=oauth;host=192.168.3.188', 'username' => 'root', 'password' => '123456']);

        // 使用Redis方式存储
        $redis = new Redis(config('redis'));

        $storage = new StorageRedis($redis);

        // Authorized example: http://swoole-chat.com/index/index/testAuthorize?response_type=code&client_id=phpweb&state=xyz
        $storage->setClientDetails('ios', '123456', 'http://swoole-chat.com', 'authorization_code password', 'read write');
        $storage->setClientDetails('android', '123456', 'http://swoole-chat.com', 'authorization_code password', 'read write');
        $storage->setClientDetails('phpweb', '123456', 'http://swoole-chat.com', 'authorization_code password', 'read write');

        $server = new Server($storage);

        // 授权码
        $server->addGrantType(new AuthorizationCode($storage));

        $this->server = $server;
    }

    /**
     * Authorized认证
     * example: http://swoole-chat.com/index/index/testAuthorize?response_type=code&client_id=phpweb&state=xyz
     * @return bool|string
     */
    public function testAuthorize() {
        $request = Request::createFromGlobals();
        $response = new Response();

        if (!$this->server->validateAuthorizeRequest($request, $response)) {
            $response->send();
            die;
        }

        if (empty($_POST)) {
            return '
                <form method="post">
                  <label>是否对testclient用户授权?</label><br />
                  <input type="submit" name="authorized" value="yes">
                  <input type="submit" name="authorized" value="no">
                </form>
            ';
        }

        $is_authorized = ($_POST['authorized'] === 'yes');
        $this->server->handleAuthorizeRequest($request, $response, $is_authorized, 1);

        if ($is_authorized) {
            $code = substr($response->getHttpHeader('Location'), strpos($response->getHttpHeader('Location'), 'code=') + 5, 40);
            return $code;
        }

        return '授权失败';
    }

    /**
     * 获取token
     * @return [type] [description]
     */
    public function testToken() {
        return $this->server->handleTokenRequest(Request::createFromGlobals())->send();
    }

    /**
     * 请求API
     * @return [type] [description]
     */
    public function testApi() {
        $request = Request::createFromGlobals();

        if (!$this->server->verifyResourceRequest($request)) {
            return $this->server->getResponse()->send();
        }

        $result = $this->server->getAccessTokenData($request);

        return json_encode(['success' => true, 'message' => '你可以访问API了', 'userId' => $result['user_id']]);
    }
}

令牌刷新

  • controller

    use OAuth2\GrantType\RefreshToken;
use OAuth2\Scope;
use OAuth2\ScopeTest;
use utils\Crypt;
use OAuth2\Server;
use OAuth2\GrantType\AuthorizationCode;
use OAuth2\GrantType\ClientCredentials;
use OAuth2\GrantType\UserCredentials;
use OAuth2\Request;
use OAuth2\Response;
use OAuth2\Storage\Redis as StorageRedis;
use cache\Redis;

header('Access-Control-Allow-Origin:*');

class Index {

    protected $server;

    public function __construct() {
        // Autoloader::register();

        // $storage = new Pdo(['dsn' => 'mysql:dbname=oauth;host=192.168.3.188', 'username' => 'root', 'password' => '123456']);

        $redis = new Redis(config('redis'));

        $storage = new StorageRedis($redis);

        $storage->setClientDetails('ios', '123456', 'http://swoole-chat.com', 'authorization_code password refresh_token', 'read write');
        $storage->setClientDetails('android', '123456', 'http://swoole-chat.com', 'authorization_code password refresh_token', 'read write');
        $storage->setClientDetails('phpweb', '123456', 'http://swoole-chat.com', 'authorization_code password refresh_token', 'read write');

        // 刷新令牌 example: curl -u phpweb:123456 http://swoole-chat.com/index/index/testToken -d 'grant_type=refresh_token&refresh_token=xxxxxx'

        $server = new Server($storage, [
            'always_issue_new_refresh_token' => true,
            'refresh_token_lifetime' => 1209600 // 14天
        ]);

        // 用户凭据
        $server->addGrantType(new UserCredentials($storage));

        // 令牌刷新
        $server->addGrantType(new RefreshToken($storage));

        $this->server = $server;
    }

    /**
     * 获取token
     * @return [type] [description]
     */
    public function testToken() {

        // 校验用户登录信息是否正确，更改username为用户ID来验证请求信息
        $_POST['username'] = 3;

        $request = Request::createFromGlobals();
        return $this->server->handleTokenRequest($request)->send();
    }

    /**
     * 请求API
     * @return [type] [description]
     */
    public function testApi() {
        $request = Request::createFromGlobals();

        if (!$this->server->verifyResourceRequest($request)) {
            return $this->server->getResponse()->send();
        }

        $result = $this->server->getAccessTokenData($request);

        return json_encode(['success' => true, 'message' => '你可以访问API了', 'userId' => $result['user_id']]);
    }
}
