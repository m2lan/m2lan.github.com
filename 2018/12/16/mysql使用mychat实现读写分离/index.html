mysql使用mychat实现读写分离

下载地址：http://dl.mycat.io/1.6-RELEASE/

环境：

中间件160服务器，写150服务器，读170服务器

150服务器

# 安装mysql服务
[root@m2lan150 ~]# yum install -y mysql-server mysql
[root@m2lan150 ~]# service mysqld restart

# 添加mycat用户
mysql> grant all on *.* to mycat@'192.168.1.160' identified by '123456';

# 配置主从关系的主服务器
[root@m2lan150 ~]# vim /etc/my.cnf
[mysqld]
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock
user=mysql
# Disabling symbolic-links is recommended to prevent assorted security risks
symbolic-links=0

log-bin=mysqllog
server-id=1
binlog-do-db=TESTDB

[mysqld_safe]
log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid

# 添加同步数据的用户
mysql> grant replication slave on *.* to slave@'192.168.1.170' identified by '123456';

[root@m2lan150 ~]# service mysqld restart

170服务器

# 安装mysql服务
[root@m2lan170 ~]# yum install -y mysql-server mysql
[root@m2lan170 ~]# service mysqld restart

# 添加mycat用户
mysql> grant all on *.* to mycat@'192.168.1.160' identified by '123456';

# 配置主从关系的从服务器
[root@m2lan170 ~]# vim /etc/my.cnf
[mysqld]
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock
user=mysql
# Disabling symbolic-links is recommended to prevent assorted security risks
symbolic-links=0

server-id=2
master-host=192.168.1.150
master-user=slave
master-password=123456
 master-connect-retry=60

[mysqld_safe]
log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid

[root@m2lan170 ~]# service mysqld restart

主从环境配好后，在150服务器创建TESTDB数据库

mysql> create database TESTDB default charset=utf8;

160服务器

# 下载mycat并解压
[root@redhat160 ~]# wget http://dl.mycat.io/1.6-RELEASE/Mycat-server-1.6-RELEASE-20161028204710-linux.tar.gz
[root@redhat160 ~]# tar xvf Mycat-server-1.6-RELEASE-20161028204710-linux.tar.gz  -C /usr/local/
# 配置mycat环境变量
[root@redhat160 ~]# vim /root/.bashrc
export MYCAT_HOME=/usr/local/mycat
export PATH=$PATH:$MYCAT_HOME/bin

[root@redhat160 ~]# source /root/.bashrc

# 下载jdk并安装
[root@redhat160 ~]# rpm -ivh jdk-8u181-linux-x64.rpm
# 配置环境变量
[root@redhat160 ~]# vim /root/.bashrc
export JAVA_HOME=/usr/java/jdk1.8.0_181-amd64
export CLASSPATH=.:$JAVA_HOME/jre/lib/rt.jar:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar
export PATH=$PATH:$JAVA_HOME/bin

[root@redhat160 ~]# source /root/.bashrc

# 配置hosts
[root@redhat160 ~]# vim /etc/hosts
192.168.1.150 m2lan150.com
192.168.1.160 redhat160
192.168.1.170 m2lan170.com

# 配置mycat
[root@redhat160 ~]# cd /usr/local/mycat/conf
[root@redhat160 conf]# cp schema.xml schema.xml.bak
[root@redhat160 conf]# vim schema.xml
<?xml version="1.0"?>
<!DOCTYPE mycat:schema SYSTEM "schema.dtd">
<mycat:schema xmlns:mycat="http://io.mycat/">

        <schema name="TESTDB" checkSQLschema="false" sqlMaxLimit="100" dataNode="dn1">
        </schema>
        <dataNode name="dn1" dataHost="localhost1" database="TESTDB" />
        <dataHost name="localhost1" maxCon="1000" minCon="10" balance="1"
                          writeType="0" dbType="mysql" dbDriver="native" switchType="-1"  slaveThreshold="100">
                <heartbeat>select user()</heartbeat>
                <writeHost host="redhat150.com" url="192.168.1.150:3306" user="mycat" password="123456">
                </writeHost>
                <writeHost host="redhat170.com" url="192.168.1.170:3306" user="mycat" password="123456"></writeHost>
        </dataHost>
</mycat:schema>

# 启动mycat，mycat监听端口为8066
[root@redhat160 conf]# mycat start

# 查看启动日志
[root@redhat160 conf]# cd ../logs/
[root@redhat160 logs]# tail -f wrapper.log
# 打印以下内容表示启动成功
 MyCAT Server startup successfully. see logs in logs/mycat.log

# 连接测试
# 这里连接的用户密码在conf/server.xml中的<root>节点配置的
[root@redhat160 logs]# mysql -uroot -p -h 192.168.1.160 -P 8066
